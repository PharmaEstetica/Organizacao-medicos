## CORREÇÃO 1: Formatação de Valores (2 casas decimais)
```typescript
function formatCurrency(value: number): string {
  return value.toLocaleString('pt-BR', {
    style: 'currency',
    currency: 'BRL',
    minimumFractionDigits: 2,
    maximumFractionDigits: 2  // SEMPRE 2 casas
  });
}

// Exemplo:
// 35.625 → "R$ 35,63" (arredonda)
// 483.897 → "R$ 483,90"
```

## CORREÇÃO 3: TOTAL dos não efetivados apenas 1x
```typescript
// Problema: está somando em cada página
// Solução: só mostrar TOTAL após o ÚLTIMO pedido não efetivado

function drawNonEffectiveSection(doc, orders, commissionPct) {
  let y = currentY;
  
  // Header da seção
  doc.text('Pedidos não efetivados', 50, y);
  y += 20;
  
  // Header da tabela
  y = drawTableHeader(doc, y, commissionPct);
  
  // Desenhar TODOS os pedidos (com quebra de página se necessário)
  for (let i = 0; i < orders.length; i++) {
    if (y > 720) {
      doc.addPage();
      y = 40;
      y = drawTableHeader(doc, y, commissionPct);  // Repetir header
      // NÃO colocar TOTAL aqui!
    }
    y = drawOrderRow(doc, y, orders[i], commissionPct);
  }
  
  // TOTAL apenas no final de TODOS os pedidos
  y += 10;
  const totalValue = orders.reduce((sum, o) => sum + o.netValue, 0);
  const totalCommission = totalValue * (commissionPct / 100);
  
  doc
    .font('Helvetica-Bold')
    .text('TOTAL', 50, y)
    .text(formatCurrency(totalValue), 280, y, { align: 'right', width: 80 })
    .text(formatCurrency(totalCommission), 450, y, { align: 'right', width: 70 });
}
```

## CORREÇÃO 4: Layout Visual com Linhas (Modelo do PDF Correto)
```typescript
// Cores e estilos do PDF de referência:

const STYLES = {
  headerBg: '#F5F5F5',      // Cinza claro para header da tabela
  lineColor: '#CCCCCC',      // Cinza para linhas divisórias
  textColor: '#000000',      // Preto para texto
  redColor: '#CC0000',       // Vermelho para COMPRAS (negativo)
  fontSize: {
    title: 12,
    header: 9,
    body: 9,
    total: 10
  }
};

// Desenhar linha horizontal
function drawLine(doc, y, startX = 40, endX = 555) {
  doc
    .strokeColor(STYLES.lineColor)
    .lineWidth(0.5)
    .moveTo(startX, y)
    .lineTo(endX, y)
    .stroke();
}

// Header da tabela com fundo cinza
function drawTableHeader(doc, y, commissionPct) {
  // Fundo cinza
  doc
    .rect(40, y, 515, 18)
    .fill(STYLES.headerBg);
  
  // Linha abaixo do header
  drawLine(doc, y + 18);
  
  // Textos do header
  doc
    .fillColor(STYLES.textColor)
    .fontSize(STYLES.fontSize.header)
    .font('Helvetica-Bold')
    .text('Data', 50, y + 5)
    .text('Status', 115, y + 5)
    .text('Valor Líquido', 200, y + 5)
    .text('Paciente', 300, y + 5)
    .text(`${commissionPct}%`, 480, y + 5);
  
  return y + 22;
}

// Linha de pedido com divisória
function drawOrderRow(doc, y, order, commissionPct) {
  const commission = order.netValue * (commissionPct / 100);
  
  doc
    .fillColor(STYLES.textColor)
    .fontSize(STYLES.fontSize.body)
    .font('Helvetica')
    .text(formatDate(order.orderDate), 50, y)
    .text(order.status, 115, y)
    .text(formatCurrency(order.netValue), 200, y, { align: 'right', width: 70 })
    .text(truncateText(order.patient, 35), 300, y)
    .text(formatCurrency(commission), 480, y, { align: 'right', width: 60 });
  
  // Linha divisória sutil
  drawLine(doc, y + 12);
  
  return y + 16;
}

// Linha de TOTAL (mais grossa, negrito)
function drawTotalRow(doc, y, label, value, commissionValue) {
  // Linha acima do total (mais grossa)
  doc
    .strokeColor('#999999')
    .lineWidth(1)
    .moveTo(40, y - 2)
    .lineTo(555, y - 2)
    .stroke();
  
  doc
    .fillColor(STYLES.textColor)
    .fontSize(STYLES.fontSize.total)
    .font('Helvetica-Bold')
    .text(label, 50, y)
    .text(formatCurrency(value), 200, y, { align: 'right', width: 70 })
    .text(formatCurrency(commissionValue), 480, y, { align: 'right', width: 60 });
  
  return y + 20;
}
```

## CORREÇÃO 5: Ordenação Cronológica
```typescript
// Ordenar pedidos por data CRESCENTE (mais antigo primeiro)

function sortOrdersByDate(orders: Order[]) {
  return [...orders].sort((a, b) => {
    return new Date(a.orderDate).getTime() - new Date(b.orderDate).getTime();
  });
}

// Aplicar antes de gerar o PDF:
const efetivadosOrdenados = sortOrdersByDate(efetivados);
const naoEfetivadosOrdenados = sortOrdersByDate(naoEfetivados);
```
